🌐 跨域代理脚本

解决 UI 自动化测试中的跨域限制问题，支持三种方案：

  • 方案 A： Node.js 单文件代理脚本（推荐，零配置）
  • 方案 B： Chrome/Edge 插件（Manifest V3）
  • 方案 C： 手动复制 DOM JSON（无需安装）

A Node.js 代理脚本

特点：

  • 单文件，零配置，体积 < 50 行
  • 自动注入 Access-Control-Allow-Origin: *
  • 支持录制模式：请求 + 响应写入 localStorage
  • 离线回放调试

使用方法：

  1. 保存下方代码为 cors-proxy.js
  2. 运行：node cors-proxy.js
  3. 访问：http://localhost:8080/proxy?url=https://target-site.com
// cors-proxy.js - 跨域代理脚本 (Node.js)
// 启动：node cors-proxy.js
// 访问：http://localhost:8080/proxy?url=https://example.com

const http = require('http');
const https = require('https');
const url = require('url');

const PORT = process.env.PORT || 8080;
const RECORD_MODE = process.env.RECORD_MODE === 'true';

console.log(`🚀 CORS Proxy 启动中... 端口：${PORT}`);
console.log(`📼 录制模式：${RECORD_MODE ? '开启' : '关闭'}`);

const server = http.createServer((req, res) => {
    const parsedUrl = url.parse(req.url, true);
    
    // 健康检查
    if (parsedUrl.pathname === '/health') {
        res.writeHead(200, { 'Content-Type': 'application/json' });
        res.end(JSON.stringify({ status: 'ok', timestamp: new Date().toISOString() }));
        return;
    }
    
    // 代理服务
    if (parsedUrl.pathname === '/proxy') {
        const targetUrl = parsedUrl.query.url;
        
        if (!targetUrl) {
            res.writeHead(400, { 'Content-Type': 'application/json' });
            res.end(JSON.stringify({ error: '缺少 url 参数' }));
            return;
        }
        
        console.log(`📡 代理请求：${targetUrl}`);
        
        const target = new URL(targetUrl);
        const protocol = target.protocol === 'https:' ? https : http;
        
        const options = {
            hostname: target.hostname,
            port: target.port || (target.protocol === 'https:' ? 443 : 80),
            path: target.pathname + target.search,
            method: req.method,
            headers: {
                ...req.headers,
                host: target.hostname
            }
        };
        
        const proxyReq = protocol.request(options, (proxyRes) => {
            // 注入 CORS 头
            res.writeHead(proxyRes.statusCode, {
                ...proxyRes.headers,
                'Access-Control-Allow-Origin': '*',
                'Access-Control-Allow-Methods': 'GET, POST, PUT, DELETE, OPTIONS',
                'Access-Control-Allow-Headers': 'Content-Type, Authorization',
                'Access-Control-Expose-Headers': '*'
            });
            
            proxyRes.pipe(res);
            
            // 录制模式：保存响应
            if (RECORD_MODE && req.method === 'GET') {
                let data = '';
                proxyRes.on('data', chunk => data += chunk);
                proxyRes.on('end', () => {
                    const record = {
                        timestamp: new Date().toISOString(),
                        url: targetUrl,
                        statusCode: proxyRes.statusCode,
                        response: data.substring(0, 50000) // 限制大小
                    };
                    console.log(`📼 录制：${targetUrl} (${data.length} bytes)`);
                    // 实际环境可写入文件或数据库
                });
            }
        });
        
        proxyReq.on('error', (e) => {
            console.error(`❌ 代理错误：${e.message}`);
            res.writeHead(502, { 'Content-Type': 'application/json' });
            res.end(JSON.stringify({ error: e.message }));
        });
        
        req.pipe(proxyReq);
        return;
    }
    
    // 前端页面（用于测试和配置）
    if (parsedUrl.pathname === '/') {
        res.writeHead(200, { 'Content-Type': 'text/html' });
        res.end(`
CORS Proxy


🌐 CORS Proxy 已就绪


使用方式：


http://localhost:${PORT}/proxy?url=https://example.com


录制模式：${RECORD_MODE ? '✅ 开启' : '❌ 关闭'} (设置 RECORD_MODE=true 启用)

`);
        return;
    }
    
    res.writeHead(404);
    res.end('Not Found');
});

server.listen(PORT, () => {
    console.log(`✅ 服务器运行在 http://localhost:${PORT}`);
});

B Chrome/Edge 插件

特点：

  • Manifest V3，兼容最新浏览器
  • Content-Script 自动注入
  • DOM 快照序列化后通过 runtime.sendMessage 抛回主应用
  • 与现有解析器 100% 兼容

安装步骤：

  1. 下载下方插件包并解压
  2. 打开 Chrome → 扩展程序 → 加载已解压的扩展程序
  3. 选择解压后的文件夹
  4. 刷新目标页面即可生效
// manifest.json
{
  "manifest_version": 3,
  "name": "CORS Helper for AI Test",
  "version": "1.0.0",
  "description": "跨域 DOM 抓取助手",
  "permissions": ["activeTab", "scripting"],
  "host_permissions": [""],
  "content_scripts": [{
    "matches": [""],
    "js": ["content-script.js"],
    "run_at": "document_end"
  }],
  "action": {
    "default_popup": "popup.html"
  }
}
// content-script.js
(function() {
    console.log('[CORS Helper] 已注入');
    
    // 监听来自主应用的消息
    window.addEventListener('message', (event) => {
        if (event.data?.type === 'CAPTURE_DOM') {
            const snapshot = {
                url: location.href,
                title: document.title,
                timestamp: new Date().toISOString(),
                dom: document.documentElement.outerHTML,
                cookies: document.cookie,
                localStorage: Object.assign({}, localStorage),
                viewport: {
                    width: window.innerWidth,
                    height: window.innerHeight
                }
            };
            
            // 发送回主应用
            event.source.postMessage({
                type: 'DOM_SNAPSHOT',
                data: snapshot
            }, event.origin);
            
            console.log('[CORS Helper] DOM 快照已发送');
        }
    });
})();

C 手动复制 DOM JSON

适用场景：

  • 无法安装插件或运行脚本
  • 临时调试需求
  • 安全限制严格的环境

操作步骤：

  1. 在目标页面打开开发者工具（F12）
  2. 在 Console 中粘贴下方代码并回车
  3. 复制输出的 JSON
  4. 在 AI Test Platform 中选择"导入 DOM JSON"
// 在目标页面 Console 中执行
(function captureDOM() {
    const snapshot = {
        url: location.href,
        title: document.title,
        timestamp: new Date().toISOString(),
        dom: document.documentElement.outerHTML,
        cookies: document.cookie,
        viewport: {
            width: window.innerWidth,
            height: window.innerHeight
        },
        elements: Array.from(document.querySelectorAll('input, button, a, select, textarea'))
            .map((el, i) => ({
                id: i,
                tag: el.tagName.toLowerCase(),
                text: el.textContent?.trim().substring(0, 50) || '',
                attributes: {
                    id: el.id,
                    name: el.name,
                    class: el.className,
                    type: el.type,
                    value: el.value,
                    href: el.href,
                    placeholder: el.placeholder
                }
            }))
    };
    
    console.log('📋 请复制以下 JSON:');
    console.log(JSON.stringify(snapshot, null, 2));
    
    // 自动复制到剪贴板
    navigator.clipboard.writeText(JSON.stringify(snapshot, null, 2))
        .then(() => console.log('✅ 已自动复制到剪贴板'))
        .catch(() => console.log('⚠️ 请手动复制上方 JSON'));
    
    return snapshot;
})();
